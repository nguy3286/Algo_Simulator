def twoSum(nums, target):
        N = len(nums)
        for i in range(N): nums[i] = (nums[i],i)
        nums.sort()
        
        l, r = 0, N-1
        while l < r:
            curr_sum = nums[l][0] + nums[r][0]
            if curr_sum == target: return [nums[l][1], nums[r][1]]
            elif curr_sum < target: l += 1
            else: r -= 1
